def verificar_caracteres_unicos(palavra):
  # Cria um conjunto a partir dos caracteres da palavra
  conjunto_caracteres = set(palavra)
  
  # Compara o tamanho do conjunto com o comprimento da palavra
  if len(palavra) == len(conjunto_caracteres):
    return "A palavra possui todos os caracteres únicos."
  else:
    return "A palavra possui caracteres repetidos."

# Exemplo de uso
palavra1 = "abcde"
palavra2 = "carrocel"

print(f"'{palavra1}': {verificar_caracteres_unicos(palavra1)}") # Saída: A palavra possui todos os caracteres únicos.
print(f"'{palavra2}': {verificar_caracteres_unicos(palavra2)}") # Saída: A palavra possui caracteres repetidos.

#crie um dicionario onde a palvra chave o o valor é o numero de vezes que ela aparece
frase = "o rato roeu a roupa do rei de roma e a rainha de raiva roeu o resto"
palavra = frase.split() #separa as palavras
contador_de_palavras ={}
for p in palavra:
    if p in contador_de_palavras:
       contador_de_palavras[p] += 1
    else:
       contador_de_palavras[p] = 1
       print(contador_de_palavras)
